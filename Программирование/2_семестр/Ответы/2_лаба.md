# //------------------------------------------------------- 1
```bash
int F1(list *p)
{
    int n;
    for (n = 0; p != NULL; p = p->next, n++);
    return n;
}
```

<div>
<b>Что делает код:</b><br>
Функция <i>F1</i> подсчитывает количество элементов в двусвязном списке и возвращает это значение.
<br><br>
<b>Построчное объяснение:</b><br>

```bash
int F1(list *p)
```

— Функция принимает указатель на первый элемент списка. p указывает на голову списка, с которой начинается обход.

<br>

```bash
int n;
```

— Объявляется переменная n, которая будет использоваться для подсчёта количества элементов в списке.

<br>

```bash
for (n = 0; p != NULL; p = p->next, n++);
```

— Цикл for начинается с n = 0 (начинаем отсчёт с нуля).

Условие: цикл выполняется до тех пор, пока не дойдём до конца списка (пока p не станет равным NULL).
p = p->next — переходим к следующему элементу списка.
n++ — увеличиваем счётчик на каждой итерации, т.е. каждый раз, когда переходим к следующему элементу списка.
<br>

```bash
return n;
```

— Возвращаем общее количество элементов, которое накопилось в переменной n.

</div>
</div>

# //------------------------------------------------------- 2
```bash
list *F2(list *ph, int v)
{
    list *q = new list;
    q->val = v;
    q->next = ph;
    ph = q;
    return ph;
}
```
<div>
<b>Что делает код:</b><br>
Функция <i>F2</i> добавляет новый элемент в начало двусвязного списка и возвращает обновлённый указатель на голову списка.
<br><br>
<b>Построчное объяснение:</b><br>

```bash
list *F2(list *ph, int v)
```

— Функция принимает два аргумента: указатель на голову списка ph и значение v, которое будет присвоено новому элементу.

<br>

```bash
list *q = new list;
```

— Создаём новый элемент списка в динамической памяти (используется оператор new, что означает работу с динамической памятью в C++).

<br>

```bash
q->val = v;
```

— Присваиваем значению нового элемента (q) переданное значение v.

<br>

```bash
q->next = ph;
```

— Указываем, что следующий элемент после нового (по указателю next) будет тот, на который указывает старая голова списка ph.

<br>

```bash
ph = q;
```

— Обновляем указатель на голову списка: теперь ph указывает на новый элемент, который стал головой списка.

<br>

```bash
return ph;
```

— Возвращаем обновлённый указатель на голову списка, который теперь указывает на только что добавленный элемент.
</div>

# //------------------------------------------------------- 3
```bash
list *F3(list *p, int n)
{
    for (; n != 0 && p != NULL; n--, p = p->next);
    return p;
}
```
<div>
<b>Что делает код:</b><br>
Функция <i>F3</i> перемещает указатель по списку на `n` элементов и возвращает указатель на новый элемент (или NULL, если список закончился).

<br><br>
<b>Построчное объяснение:</b><br>

```bash
list *F3(list *p, int n)
```

— Функция принимает два аргумента: указатель на элемент списка p и количество шагов n, на которое нужно переместить указатель.

<br>

```bash
for (; n != 0 && p != NULL; n--, p = p->next);
```

— Цикл выполняется до тех пор, пока:

n != 0 (пока не сделали все необходимые шаги)
p != NULL (пока не дошли до конца списка)
На каждой итерации:
n-- — уменьшаем количество оставшихся шагов
p = p->next — перемещаем указатель на следующий элемент списка
<br>

```bash
return p;
```

— Возвращаем указатель на новый элемент списка, на который переместились. Если список закончился, возвращается NULL.
<p></p>
</div>

```bash
list *F4(list *ph, int v)
{
    list *p, *q = new list;
    q->val = v;
    q->next = NULL;
    if (ph == NULL) return q;
    for (p = ph; p->next != NULL; p = p->next);
    p->next = q;
    return ph;
}
```
<div>
  <b>Что делает код:</b><br>
  Функция <i>F4</i> добавляет новый элемент в конец двусвязного списка и возвращает указатель на голову списка. Если список пуст, новый элемент становится первым и единственным в списке.

</div>

```bash
list *F5(list *ph, int n)
{
    list *q, *pr, *p;
    for (p = ph, pr = NULL; n != 0 && p != NULL; n--, pr = p, p = p->next);
    // © Романов Е.Л. Беседы о программировании 13.
    if (p == NULL) return ph;
    if (pr == NULL) {
        q = ph;
        ph = ph->next;
    } else {
        q = p;
        pr->next = p->next;
    }
    delete q;
    return ph;
}
```
<div>
<div>
  <b>Что делает код:</b><br>
  Функция <i>F5</i> удаляет элемент из двусвязного списка, который находится на позиции `n` (счёт с 1). Если элемент с таким индексом найден, то он удаляется, и возвращается указатель на голову списка. Если элемент не найден (например, `n` больше длины списка), то возвращается оригинальный указатель на голову списка.

```bash
int F6(list *p)

    int n;
    list *q;
    if (p == NULL) return 0;
    for (q = p, p = p->next, n = 1; p != q; p = p->next, n++);
    return n;
}
```
<div>
  <b>Что делает код:</b><br>
  Функция <i>F6</i> подсчитывает количество элементов в циклическом списке. Если список пустой (NULL), функция возвращает 0. В противном случае она выполняет обход списка, начиная с головы, и возвращает количество элементов до первого повторения (когда указатель снова вернется к начальной позиции).

</div>

```bash
list *F7(list *p, int v)
{
    list *q;
    q = new list;
    q->val = v;
    q->next = q->prev = q;
    if (p == NULL) p = q;
    else {
        q->next = p;
        q->prev = p->prev;
        p->prev->next = q;
        p->prev = q;
        p = q;
    }
    return p;
}
```
<div>
  <b>Что делает код:</b><br>
  Функция <i>F7</i> добавляет новый элемент в начало циклического двусвязного списка. Если список пустой (NULL), новый элемент становится единственным в списке. В противном случае новый элемент вставляется в начало списка, а указатели на соседей обновляются, сохраняя циклическую структуру.

</div>

```bash
list *F8(list *ph)
{
    list *q, *out, *p, *pr;
    out = NULL;
    while (ph != NULL) {
        q = ph;
        ph = ph->next;
        for (p = out, pr = NULL; p != NULL && q->val > p->val; pr = p, p = p->next);
        if (pr == NULL) {
            q->next = out;
            out = q;
        } else {
            q->next = p;
            pr->next = q;
        }
    }
    return out;
}
```
<div>
  <b>Что делает код:</b><br>
  Функция <i>F8</i> сортирует элементы двусвязного списка по возрастанию, используя метод вставки. Каждый элемент извлекается из исходного списка и вставляется в отсортированную часть списка таким образом, чтобы поддерживался порядок. В итоге возвращается указатель на отсортированный список.

</div>

```bash
list *F9(list *pp, int n)
{
    list *q;
    for (q = pp; n != 0; q = q->next, n--);
    if (q->next == q) {
        delete q;
        return NULL;
    }
    if (q == pp) pp = q->next;
    q->prev->next = q->next;
    q->next->prev = q->prev;
    delete q;
    return pp;
}
```
<div>
  <b>Что делает код:</b><br>
  Функция <i>F9</i> удаляет элемент из двусвязного списка, начиная с позиции `n`. Если элемент является единственным в списке, он удаляется, и список становится пустым. В случае удаления другого элемента, ссылки на соседние элементы обновляются, чтобы сохранить целостность списка. Возвращается указатель на голову списка.

</div>

```bash
list *F11(list *ph, int v)
{
    list *q = new list;
    q->val = v;
    q->next = q->prev = q;
    if (ph == NULL) return q;
    list *p = ph;
    do {
        if (v < p->val) break;
        p = p->next;
    } while (p != ph);
    q->next = p;
    q->prev = p->prev;
    p->prev->next = q;
    p->prev = q;
    if (ph->val > v) ph = q;
    return ph;
}
```
<div>
  <b>Что делает код:</b><br>
  Функция <i>F11</i> вставляет новый элемент в циклический двусвязный список, сохраняя порядок сортировки. Если список пустой, новый элемент становится единственным в списке. Если элемент нужно вставить в середину списка, все ссылки на соседние элементы обновляются, а голова списка изменяется, если это необходимо.

</div>

```bash
void F12(list *&ph, int v, int n)
{
    list *q = new list;
    int n0 = n;
    q->val = v;
    q->next = q->prev = q;
    if (ph == NULL) {
        ph = q;
        return;
    }
    list *p;
    for (p = ph; n-- != 0; p = p->next);
    q->next = p;
    q->prev = p->prev;
    p->prev->next = q;
    p->prev = q;
    if (n0 == 0) ph = q;
}
```
<div>
  <b>Что делает код:</b><br>
  Функция <i>F12</i> вставляет новый элемент в двусвязный список на позицию `n`. Если список пустой, новый элемент становится единственным в списке. В противном случае элемент вставляется в нужное место, и ссылки на соседние элементы обновляются. Если вставка происходит в начало списка, обновляется указатель на голову.

</div>
